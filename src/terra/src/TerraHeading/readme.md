<!-- description -->

A heading displays important text at four different levels. Use the levels to indicate the heading's importance.

<!-- /description -->

<!-- examples -->

## Default

```jsx
<Heading level={1}>User administration</Heading>
```

<!-- /examples -->

<!-- best-practices -->

<h2>Don't</h2>

<ul>
    <li>Don't use the heading for normal text.</li>
</ul>
